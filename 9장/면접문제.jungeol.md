# 면접문제 Jungeol
#### 1.2 순열 확인: 문자열 두 개가 주어졌을 때 이 둘이 서로 순열 관계에 있는지 확인하는 메서드를 작성하라.

```javascript
'use strict'
function solution (s1, s2) {
    if (s1.length !== s2.length) {
        return false
    }
    
    const m1 = getCountMap(s1)
    const m2 = getCountMap(s2)

    for (let pair of m1) {
        const key = pair[0]
        const value = pair[1]

        if (m2.get(key) !== value) {
            return false
        }
    }

    return true
}

function getCountMap (A) {
    const m = new Map()
    for (let c of A) {
        if (m.has(c)) {
            m.set(c, m.get(c) + 1)
        } else {
            m.set(c, 1)
        }
    }
    return m
}
```
