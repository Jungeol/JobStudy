### 1.2 순열 확인: 
문자열 두 개가 주어졌을 때 이 둘이 서로 순열 관계에 있는지 확인하는 메서드를 작성하라.

해법: 문자열에 포함된 문자의 출현횟수가 같은지 검사
```javascript
function solution (s1, s2) {
    if (s1.length !== s2.length) {
        return false
    }
    
    const m1 = getCountMap(s1)
    const m2 = getCountMap(s2)

    // 문자열별로 count를 체크한후 틀리다면 false
    for (let pair of m1) {
        const key = pair[0]
        const value = pair[1]

        if (m2.get(key) !== value) {
            return false
        }
    }

    return true
}

// 문자열 별로 count를 가진 Map 반환
function getCountMap (A) {
    const m = new Map()
    for (let c of A) {
        if (m.has(c)) {
            m.set(c, m.get(c) + 1)
        } else {
            m.set(c, 1)
        }
    }
    return m
}
```

### 1.5 하나 빼기: 
문자열을 편집하는 방법에는 세가지 종류가 있다. 문자 삽입, 문자 삭제, 문자 교체, 문자열 두 개가 주어졌을 때, 문자열을 같게 만들기 위한 편집 횟수가 1회 이내인지 확인하는 함수를 작성하라.

```javascript
function solution (s1, s2) {
    // s1과 s2간의 length 차이가 1 보다 클경우 무조건 false
    const diff = s1.length - s2.length
    if (diff > 1 || diff < -1) {
        return false
    }

    // 큰 문자열과 작은 문자열로 재정의
    const short = diff > 0 ? s2 : s1
    const long = diff > 0 ? s1 : s2
    // 틀린 문자를 빼는지, 교체하는지 길이로 정함
    const add = long.length - short.length

    let count = 0
    let si = 0
    let li = 0
    
    // 어느 한쪽의 문자열이 끝날때까지 
    // 루프를 돌며 다른 문자의 수를 체크함
    while (si < short.length && li < long.length) {
        if (short[si] !== long[li]) {
            si -= add
            count += 1
        }
        si += 1
        li += 1
    }

    // 남은 문자열의 수룰 count에 더함
    count += Math.abs((long.length - 1 - li) - (short.length - 1 - si))
    
    return count === 1
}
```

### 1.8 O 행렬: 
M x N 행렬의 한 원소가 0일경우, 해당 원소가 속한 행과 열의 모든 원소를 0으로 설정하는 알고리즘을 작성하라.

```javascript
function solution (matrix) {
    const xSet = new Set()
    const ySet = new Set()
    
    // 0인 주소를 찾아 xSet, ySet에 넣음
    for (let x in matrix) {
        for (let y in matrix[x]) {
            if (matrix[x][y] === 0) {
                xSet.add(x)
                ySet.add(y)
            }
        }
    }

    // 찾은 x 행을 전부 0으로 변경
    for (let x of xSet) {
        for (let i = 0; i < matrix[x].length; i++) {
            matrix[x][i] = 0
        }
    }

    // 찾은 y 열을 전부 0으로 변경
    for (let y of ySet) {
        for (let i = 0; i < matrix.length; i++) {
            matrix[i][y] = 0
        }
    }

    return matrix
}
```

## 02.연결리스트
단방향 LikedList 구현
```javascript
class Node {
    constructor(value) {
        this.data = value
        this.next = null
    }

    appendToTail(value) {
        var end = new Node(value)
        var n = this
        while (n.next !== null) {
            n = n.next
        }
        n.next = end
    }

    deleteNode(head, value) {
        var n = head
    
        if (head.data === value) {
            return head.next
        }
    
        while (n.next !== null) {
            if (n.next.data === value) {
                n.next = n.next.next
                return head
            }
            n = n.next
        }
        return head
    }
}

module.exports = Node
```

### 2.2 뒤에서 k번째 원소 구하기: 
단방향 연결리스트가 주어졌을때 뒤에서 k번째 원소를 찾는 알고리즘을 구현하라.

```javascript
function solution(head, k) {
    let p1 = head
    let p2 = head
    // p2에 k번째 node를 할당
    for (let i = 0; i < k; i++) {
        if (p2.next === null) {
            return null
        } else {
            p2 = p2.next
        }
    }

    // p2가 끝에 도달하게 함
    while(p2.next !== null) {
        p1 = p1.next
        p2 = p2.next
    }

    // p2보다 k번째 뒤에 있는 p1을 return
    return p1
}
```

### 2.5 리스트의 합: 
연결리스트로 숫자를 표현할 때 각 노드가 자릿수하나를 가리키는 방식으로 표현할 수 있다. 각 숫자는 역순으로 배열되어 있는데, 첫 번째 자릿수가 리스트의 맨앞에 위치하도록 배열된다는 뜻이다. 이와 같은 방식으로 표현된 숫자 두 개가 있을 때, 이 두수를 더하여 그 합을 연결리스트로 반환하는 함수를 작성하라.

```javascript
function solution(head1, head2) {
    let n1 = head1
    let n2 = head2

    let up = 0 // 올림수 저장
    
    while (n1 !== null) {
        // 각각 자리수와 올림수를 더함
        const value = n1.data + n2.data + up
        // n1의 노드에 값을 저장한 후 올림수를 할당
        n1.data = value % 10
        up = value >= 10 ? 1 : 0
        
        // n1과 n2 중 하나라도 next가 있으면 0할당
        if (n1.next === null && n2.next !== null) {
            n1.appendToTail(0)
        }
        if (n1.next !== null && n2.next === null) {
            n2.appendToTail(0)
        }
        n1 = n1.next
        n2 = n2.next
    }

    // 마지막 올림수가 있으면 마지막에 붙임
    if (up > 0) {
        head1.appendToTail(up)
    }

    return head1
}
```

### 2.8 루프 발견:
순환 연결리스트(circular linked list)가 주어졌을 때, 순환되는 부분의 첫째 노드를 반환하는 알고리즘을 작성하라. 순환 연결리스트란 노드의 next 포인터가 앞선 노드들 가운데 어느 하나를 가리키도록 설정되어 있는, 엄밀히 말해서 변질된 방식의 연결리스트를 의미한다.

1. 직접 푼 방법
    - HashMap을 이용해 Node들을 저장
    - 새로운 노드로 진행할때 마다 체크
```javascript
function solution(head) {
    const map = new Map()
    let n = head
    while (n.next !== null) {
        map.set(n, true)
        if (map.has(n.next)) {
            return true
        }
        n = n.next
    }
    return false
}
```

2. 해법(FastRunner/SlowRunner 방법)
    - 두 표인터 FastPointer와 SlowPointer를 만든다
    - FastPointer는 한번에 두 노드, SlowPointer는 한번에 한 노드씩 움직인다.
    - 두 포인터가만나면, SlowPointer는 LinkedListHead로 옮기고, FastPointer는 그 자리에 그대로 둔다.
    - SlowPlinter와 FastPointer를 한 번에 한 노드씩 움직인다. 이 둘이 만나는 지점을 반환한다.
```javascript
function solution (head) {
    let slow = head
    let fast = head

    // 만나는 지점을 찾는다. 연결리스트 안에서 LOOP_SIZE-k만큼 들어간 상태이다
    while (fast !== null && fast.next != null) {
        slow = slow.next
        fast = fast.next
        if (slow === fast) { // 충돌
            break
        }
    }

    // 에러 체크, 만나는 지점이 없다면 루프도 없다.
    if (fast === null || fast.next === null) {
        return null
    }

    // slow를 head로 옮기고, fast는 그대로 둔다. 이 둘은 루프 시작 지점에서 k만큼
    // 떨어져있다. 이 둘이 같은 속도로 움직인다면 시작 지점에서 만나게 된다
    slow = head
    while (slow != fast) {
        slow = slow.next
        fast = fast.next
    }

    // 둘 다 루프의 시작 지점을 가리킨다
    return fast
}
```

### 3.2 스택 Min:
기본적인 push와 pop 기능이 구현된 스택에서 최솟값을 반환하는 min 함수를 추가하려고 한다. 어떻게 설계할 수 있겠는가? push, pop, min 연산은 모두 O(1) 시간에 동작해야 한다.

```javascript
class Node {
    constructor(data) {
        this.data = data;
        // Node class에 min을 정의
        this.min = null;
        this.next = null;
    }
}

class Stack {
    constructor() {
        this.top = null;
    }

    push(data) {
        const node = new Node(data);
        // Node에 Min을 저장한다.
        node.min = this.top !== null ? Math.min(this.top.data, node.data) : data;
        node.next = this.top;
        this.top = node;
    }

    pop() {
        if (this.top === null) { throw new Error('stack is empty.'); }
        const data = this.top.data;
        this.top = this.top.next;
        return data;
    }

    min() {
        if (this.top === null) { throw new Error('stack is empty.'); }
        return this.top.min;
    }
}
```

### 3.5 스택정렬:
가장 작은 값이 위로 오도록 스택을 정렬하는 프로그램을 작성하라. 추가적으로 하나 정도의 스택은 사용해도 괜찮지만, 스택에 보관된 요소를 배열 등의 다른 자료구조로 복사할 수는 없다. 스택은 push, pop, peek, isEmpty의 네가지 연산을 제공해야 한다.

```javascript
function sort(s) {
    const r = new Stack();
    while(!s.isEmpty()) {
        let temp = s.pop();
        while(!r.isEmpty() && temp > r.peek()) {
            s.push(r.pop());
        }
        r.push(temp);
    }

    while(!r.isEmpty()) {
        s.push(r.pop());
    }

    return s;
}
```

## 4. 트리와 그래프
### 4.2 최소트리:
오름차순으로 정렬된 배열이 있다. 이 배열 안에 들어 있는 원소는 정수이며 중복된 값이 없다고 했을 때 높이가 최소가 되는 이진 탐색 트리를 만드는 알고리즘을 작성하라.

### 4.4 균형확인:
이진 트리가 균형 잡혀있는지 확인하는 함수를 작성하라. 이 문제에서 균형잡흔 트리란 모든 노드에 대해서 왼쪽 부분 트리의 높이와 오른쪽 부분 트르의 높이의 차이가 최대 하나인 트리를 의미한다.

### 4.6 후속자: 
이진 탐색 트리에서 주어진 노드의 '다음' 노드(중위 후속자(in-order successor))를 찾는 알고리즘을 작성하라. 각 노드에는 부모 노드를 가리키는 링크가 존재한다고 가정하자.

### 4.8 첫번째 공통 조상:
이진 트리에서 노드 두 개가 주어졌을 때 이 두 노드의 첫번째 공통 조상을 찾는 알고리즘을 설계하고 그 코드를 작성하라. 자료구조내에 추가로 노드를 저장해 두면 안 된다. 반드시 이진 탐색 트리일 필요는 없다.

### 4.10 하위 트리 확인:
두 개의 커다란 이진트리 T1과 T2가 있다고 하자. T1이 T2보다 훨씬 크다고 했을 때, T2가 T1의 하위 트리(subtree)인지 판별하는 알고리즘을 만들라. T1 안에 있는 노드 n의 하위 트리가 T2와 동일하면, T2는 T1의 하위트리다. 다시말해, T1에서 노트 n의 아래쪽을 끊어 냈을 때 그 결과가 T2와 동일해야한다.

### 4.12 합의 경로:
각 노드의 값이 정수(음수 및 양수)인 이진 트리가 있다. 이때 정수의 합이 특정 값이 되도록 하는 경로의 개수를 세려고 한다. 경로가 꼭 루트에서 시작해서 말단 노드에서 끝날 필요는 없지만 반드시 아래로 내려가야 한다. 즉, 부모 노드에서 자식노드로만 움직일 수 있다. 알고리즘을 어떻게 설계할 것인가?
